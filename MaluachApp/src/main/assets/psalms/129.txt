{א} שִׁ֗יר הַֽמַּ֫עֲל֥וֹת רַ֭בַּת צְרָר֣וּנִי מִנְּעוּרַ֑י יֹֽאמַר-נָ֝א יִשְׂרָאֵֽל: {ב} רַ֭בַּת צְרָר֣וּנִי מִנְּעוּרָ֑י גַּ֝ם לֹא-יָ֥כְלוּ לִֽי: {ג} עַל-גַּ֭בִּי חָרְשׁ֣וּ חֹרְשִׁ֑ים הֶ֝אֱרִ֗יכוּ (למענותם) לְמַעֲנִיתָֽם: {ד} יְהוָ֥ה צַדִּ֑יק קִ֝צֵּ֗ץ עֲב֣וֹת רְשָׁעִֽים: {ה} יֵ֭בֹשׁוּ וְיִסֹּ֣גוּ אָח֑וֹר כֹּ֝֗ל שֹׂנְאֵ֥י צִיּֽוֹן: {ו} יִ֭הְיוּ כַּחֲצִ֣יר גַּגּ֑וֹת שֶׁקַּדְמַ֖ת שָׁלַ֣ף יָבֵֽשׁ: {ז} שֶׁלֹּ֤א מִלֵּ֖א כַפּ֥וֹ קוֹצֵ֗ר וְחִצְנ֥וֹ מְעַמֵּֽר: {ח} וְלֹ֤א אָֽמְר֨וּ | הָעֹבְרִ֗ים בִּרְכַּֽת-יְהוָ֥ה אֲלֵיכֶ֑ם בֵּרַ֥כְנוּ אֶ֝תְכֶ֗ם בְּשֵׁ֣ם יְהוָֽה:
